<script>

let data = JSON.parse(localStorage.getItem("receipts")) || [];

function save(){

let n = document.getElementById("name").value;
let a = document.getElementById("amount").value;

if(!n || !a){
alert("Fill fields");
return;
}

let id = Date.now();

data.push({id,n,a,status:"Pending"});
localStorage.setItem("receipts", JSON.stringify(data));

show();
}

function show(){

let list = document.getElementById("list");
list.innerHTML="";
let total=0;

data.forEach((r,i)=>{

total += Number(r.a);

list.innerHTML += `
<div class="card">
<b>${r.n}</b> - $${r.a}<br>
Status: ${r.status}<br><br>

<button class="pay" onclick="pay(${i})">Mark Paid</button>
<button class="print" onclick="printOne(${r.id},'${r.n}','${r.a}','${r.status}')">Print</button>
<button class="whatsapp" onclick="sendWhats(${r.id},'${r.n}','${r.a}','${r.status}')">WhatsApp</button>

</div>
`;
});

document.getElementById("total").innerText = total;
}

function pay(i){
data[i].status="Paid";
localStorage.setItem("receipts", JSON.stringify(data));
show();
}

function sendWhats(id,n,a,s){

let fecha = new Date().toLocaleDateString();

let mensaje =
`🚗 Car Studio

🧾 Invoice #${id}
📅 ${fecha}

📍 20 Prescot St, Jersey City NJ 07304
📞 201-658-7295

👤 ${n}
💰 $${a}
📌 ${s}

🙏 Thank you!`;

let url = "https://wa.me/?text=" + encodeURIComponent(mensaje);

window.open(url);
}
